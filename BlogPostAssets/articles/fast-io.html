读输优化


今天写了一个不错的快读，跟大家分享。

废话不多说，代码如下：


#ifndef _FASTIO_H_
#define _FASTIO_H_

#include <cstdio>
#include <cctype>

namespace fastio {
    /* Fast In Stream Module */
    class FastIn {
    private:
        template<typename T>
        inline void readSingle(T& x) {
            x = 0;
            int f = 1;
            char ch = getchar();
            while (!isdigit(ch) && ch != '-')
                ch = getchar();
            if (ch == '-')
                f = -1,
                ch = getchar();
            while (isdigit(ch))
                x = x * 10 + ch - '0',
                ch = getchar();
            x *= f;
        }
    public:
        template<typename T>
        FastIn& operator>>(T& x) {
            readSingle(x);
            return *this;
        }
    };


    /* Fast Out Stream Module */
    class FastOut
    {
    private:
        inline void writeSingle(int x) {
            printf("%d", x);
        }

        inline void writeSingle(long long x) {
            printf("%lld", x);
        }

        inline void writeSingle(char ch) {
            printf("%c", ch);
        }

        inline void writeSingle(char* str) {
            printf("%s", str);
        }

        inline void writeSingle(const char* str) {
            printf("%s", str);
        }

        inline void writeSingle(string str) {
            char* buffer = new char[str.size() + 1];
            for (register int i = 0; i < str.size(); i ++)
                buffer[i] = str[i];
            buffer[str.size()] = 0;
            printf("%s", buffer);
            delete buffer;
        }

        template<typename T>
        inline void writeSingle(vector<T> v) {
            for (register int i = 0; i < v.size(); i ++)
                writeSingle(v[i]),
                writeSingle(" ");
            writeSingle("\n");
        }
    public:
        template<typename T>
        FastOut& operator<<(T x) {
            writeSingle(x);
            return *this;
        }
    };

    /* Instantiation of Standard FastIn and FastOut stream */
    FastIn  fastin;
    FastOut fastout;
}

#endif

使用方法如下：


#include <cstdio>
#include "fastio.h"
using namespace std;

int main()
{
    int a, b, c;
    fastio::fastin >> a >> b >> c;
    //可使用using namespace fastio;
    //然后用fastin >> a >> b >> c;
    printf("%d %d %d\n", a, b, c);
    return 0;
}

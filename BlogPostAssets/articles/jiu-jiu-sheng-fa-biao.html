九九乘法表

无聊小程序系列


无聊的时候写了一个打印中文九九乘法表的程序

（真的没什么卵用）

没什么好说的，直接上代码吧：

（话说这里为什么还要加优化）

		  
#include <bits/stdc++.h>

std::string num[10] = {"", "一", "二", "三", "四", "五", "六", "七", "八", "九"};

inline int printNum(register int x)
{
	register int sum;
	if (x >= 10) std::cout << num[x / 10] << "十" << num[x % 10], sum = x % 10 ? 5 : 4;
	else std::cout << num[x], sum = 4;
	return sum;
}

int main()
{
	for (register int i = 1; i <= 9; i ++)
	{
		for (register int j = 1; j <= i; j ++)
		{
			register int x = i * j;
			printNum(j);
			printNum(i);
			if (x < 10) std::cout << "得";
			for (register int k = printNum(x); k <= 5; k ++) std::cout << "  ";
		}
		std::cout << std::endl;
	}
	return 0;
}
